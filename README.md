# Streamlit

<pre>
#建立環境
python -m venv streamlit_env

#window環境
streamlit_env/Scripts/activate
pip install -r requirements.txt

#mac環境
source streamlit_env/bin/activate
pip3 install -r requirements.txt

#輸入以下啟動環境
streamlit run first_app.py
